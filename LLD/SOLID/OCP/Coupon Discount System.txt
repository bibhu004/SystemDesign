public interface Coupons{
        public double applyDiscount(double amount);
}


public class FlatCoupons extends Coupons{

    @Override
    public double applyDiscount(double amount){
        
        return amount - 100;
    }
}


public class PercentageCoupons extends Coupons{

    @Override
    public double applyDiscount(double amount){
        
        return amount - (amount%10);
    }
}
